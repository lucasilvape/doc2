    def buscar_bilhetes_por_data_criacao_implementacao(
    ambiente: Literal['PRODUCAO', 'HOMOLOGACAO', 'ANALYTICS'], 
    collection: str,
    qde_dias_retroativos: int = None

):
    db, client = abrir_conexao_mongo(ambiente=ambiente)

    pipeline_bilhetes = []

    #qde_dias_retroativos informado, filtrar por data_modificacao
    if qde_dias_retroativos is not None:
        data_limite = datetime.today() - timedelta(days=qde_dias_retroativos)
        pipeline_bilhetes.append({
            '$match': {
                '_modified': {'$gte': data_limite}
            }
        })

    pipeline_bilhetes.append({
        '$group': {
            '_id': {
                '$dateToString': {
                    'format': '%Y-%m-%d',
                    'date': '$_created'
                }
            },
            'uuid_bilhete_binary': {'$push': '$_id'},
            'data_modificacao': {'$push': {
                '$dateToString': {
                    'format': '%Y-%m-%d',
                    'date': '$_modified'
                }
            }}
        }
    })

    df = list(db[collection].aggregate(pipeline_bilhetes))
    client.close()

    df = pd.DataFrame(df)
    for dd in tqdm(df.index, desc='Processando bilhetes por data de criação implementação'):

        uuid_bilhetes = df['uuid_bilhete_binary'][dd]
        data_modificacao = df['data_modificacao'][dd]
        data = df.iloc[dd, 0]

        zz = zip(uuid_bilhetes, data_modificacao)

        df2 = pd.DataFrame(zz, columns=['uuid_bilhete_binary', 'data_modificacao'])
        df2[['data_criacao']] = data

        if dd == 0:
            df_bilhetes = df2
        else:
            df_bilhetes = pd.concat([df_bilhetes, df2], ignore_index=True)

    df_bilhetes['uuid_bilhete'] = df_bilhetes['uuid_bilhete_binary'].apply(converter_bytes_para_uuid)
    df_bilhetes = df_bilhetes[['data_criacao', 'data_modificacao', 'uuid_bilhete', 'uuid_bilhete_binary']]
    df_bilhetes['origem'] = 'IMPLEMENTACAO'
    return df_bilhetes

    Funcionalidades:

    • Agregação MongoDB com $group por data de criação
    • Retorna UUIDs em formato binario
    • Calcula data de modificação por registro
    • Marca origem como 'IMPLEMENTACAO'

    gerar_base_para_busca(...)

    Orquestra estratégias de extração com suporte a múltiplos tipos de atualização.

    Assinatura: 

    def gerar_base_para_busca(ambiente:str,qde_dias_retroativos:int,percentual_modificacao_incremental:int,tp_atualizacao:Literal['DIARIO','INCREMENTAL','FULL','DINAMICO'],nm_collection_implementacao, nm_collection_pacelas,data_criacao=None) -> Tuple[str,pd.DataFrame]:

    '''
    ambiente: str
        Ambiente de conexão com o MongoDB, ex: 'PRODUCAO', 'HOMOLOGACAO', etc.
    qde_dias_retroativos: int   
        Quantidade de dias retroativos para considerar modificações.        
    percentual_modificacao_incremental: int 
        Percentual de modificações para considerar a atualização como incremental.
    tp_atualizacao: Literal['DIARIO','INCREMENTAL','FULL','DINAMICO']
        DIARIO: Deverá receber uma data no padrão 'yyyy-mm-dd' para buscar os bilhetes daquele dia.
        INCREMENTAL: Considera as emissões nos ultimos dias(qde) , a quantidade de dias deverá ser informada na variável qde_dias_retroativos.
        FULL: Considera todos os bilhetes, independente de modificações.
        DINAMICO: Determina o tipo de atualização com base no percentual de modificações
                Se o percentual de modificações exceder o limite informado na variável "percentual_modificacao_incremental" será retornado a base full,
                caso contrário retornará somente os registros que foram modificados nos ultimos dias, a quantidade de dias deverá respeitar a variável qde_dias_retroativos .

    '''
    INCREMENTAL_CYBER = False

    if not tp_atualizacao in ['DIARIO','INCREMENTAL','FULL','DINAMICO']:
        raise ValueError("tp_atualizacao deve ser 'DIARIO', 'INCREMENTAL', 'FULL' ou 'DINAMICO'.")
    if tp_atualizacao == 'DIARIO' and not data_criacao:
        raise ValueError("Para tp_atualizacao 'DIARIO', deve ser fornecida uma data de criação no formato 'yyyy-mm-dd'. A data deve ser informada na variável data_criacao.")
    if tp_atualizacao == 'DIARIO':
        try:
            data = pd.to_datetime(data_criacao, format='%Y-%m-%d')
        except:
            raise ValueError("data_criacao deve estar no formato 'yyyy-mm-dd'.")


    if nm_collection_implementacao  == 'implementations.cyberpicpay' and tp_atualizacao == 'DINAMICO':
        print('Implementacao dp Picpay Cyber')
        INCREMENTAL_CYBER = True
        df_implementacao = buscar_bilhetes_por_data_criacao_implementacao(ambiente=ambiente,collection=nm_collection_implementacao,qde_dias_retroativos=qde_dias_retroativos)
    else:
        #print('FULL CYBER')
        df_implementacao = buscar_bilhetes_por_data_criacao_implementacao(ambiente=ambiente,collection=nm_collection_implementacao)

    if nm_collection_pacelas == 'implementations.installmentsgeneric' and tp_atualizacao == 'DINAMICO':
        INCREMENTAL_CYBER = True
        df_parcelas = buscar_bilhetes_por_data_criacao_parcelas(ambiente=ambiente,collection=nm_collection_pacelas,qde_dias_retroativos=qde_dias_retroativos)
    else:
        #print('FULL CYBER')
        df_parcelas = buscar_bilhetes_por_data_criacao_parcelas(ambiente=ambiente,collection=nm_collection_pacelas)


    dfs = [df_implementacao,df_parcelas]

    if INCREMENTAL_CYBER:
        print(f'########################  TIPO ATUALIZACAO DINAMICA -- INCREMENTAL -- PICPAY CYBER  ########################')

        return 'INCREMENTAL' ,  df_implementacao,df_parcelas

    for df in dfs:
        df['data_modificacao'] = pd.to_datetime(df['data_modificacao'])
        df['AGING_MODIFICACAO'] = (pd.to_datetime('today') - df['data_modificacao']).dt.days
        df['AGING_MODIFICACAO'] = pd.to_numeric(df['AGING_MODIFICACAO'], errors='coerce')
        df.sort_values('AGING_MODIFICACAO', ascending=True,inplace=True)

    df_parcelas.drop_duplicates(subset='uuid_parcela',inplace=True)
    df_implementacao.drop_duplicates(subset='uuid_bilhete',inplace=True)

    if tp_atualizacao == 'FULL':
        print(f'########################  TIPO ATUALIZACAO FULL  ########################')
        return 'FULL',df_implementacao,df_parcelas

    elif tp_atualizacao == 'DIARIO':
        print(f'########################  TIPO ATUALIZACAO DIARIO  ########################')
        df_implementacao = df_implementacao[df_implementacao['data_criacao'] == data_criacao]
        df_parcelas = df_parcelas[df_parcelas['data_criacao'] == data_criacao]
        return 'DIARIO',df_implementacao,df_parcelas

    elif tp_atualizacao == 'INCREMENTAL':
        print(f'########################  TIPO ATUALIZACAO INCREMENTAL  ########################')
        dt_max_imp = max(df_implementacao['data_criacao'])
        nova_data = datetime.strptime(dt_max_imp,'%Y-%m-%d') - timedelta(days=qde_dias_retroativos)


        df_implementacao['data_criacao2'] = pd.to_datetime(df_implementacao['data_criacao'], errors='coerce')
        df_implementacao = df_implementacao[df_implementacao['data_criacao2'] >= nova_data]
        df_implementacao.drop(columns=['data_criacao2'], inplace=True)


        dt_max_parcelas = max(df_parcelas['data_criacao'])
        nova_data = datetime.strptime(dt_max_parcelas,'%Y-%m-%d') - timedelta(days=qde_dias_retroativos)


        df_parcelas['data_criacao2'] = pd.to_datetime(df_parcelas['data_criacao'], errors='coerce')
        df_parcelas = df_parcelas[df_parcelas['data_criacao2'] >= nova_data]
        df_parcelas.drop(columns=['data_criacao2'], inplace=True)



        return 'INCREMENTAL',df_implementacao,df_parcelas

    incremental_implementacao= df_implementacao.loc[df_implementacao['AGING_MODIFICACAO'] <= qde_dias_retroativos,'uuid_bilhete'].dropna().unique()
    incremental_parcelas= df_parcelas.loc[df_parcelas['AGING_MODIFICACAO'] <= qde_dias_retroativos,'uuid_parcela'].dropna().unique()

    percentual_modificacao = 0 if df_parcelas.empty else round((len(incremental_parcelas) / df_parcelas.shape[0]) * 100, 0)

    print(f'########################  Tamanho da base de implementacao: {df_implementacao.shape[0]}   ########################')
    print(f'########################  Tamanho da base de parcelas: {df_parcelas.shape[0]}   ########################')
    print(f'########################  Percentual de bilhetes que sofreram modificação: {percentual_modificacao}   ########################')

    if tp_atualizacao == 'DINAMICO':

        if percentual_modificacao >= percentual_modificacao_incremental:
            print(f'########################  TIPO ATUALIZACAO DINAMICA  -- FULL  -- LIMITE PERCENTUAL MODIFICAÇÃO: {percentual_modificacao_incremental}   ########################')
            return 'FULL',df_implementacao,df_parcelas
        else:
            print(f'########################  TIPO ATUALIZACAO DINAMICA -- INCREMENTAL -- LIMITE PERCENTUAL MODIFICAÇÃO: {percentual_modificacao_incremental}  ########################')

            df_incremental_implementacao = df_implementacao[df_implementacao.uuid_bilhete.isin(incremental_implementacao)]
            df_incremental_parcela = df_parcelas[df_parcelas.uuid_parcela.isin(incremental_parcelas)]
            return 'INCREMENTAL' ,  df_incremental_implementacao,df_incremental_parcela

    Estratégias Suportadas:

    Tipo Descrição Quando Usar
    FULL Retorna todos os registros Recarga completa
    DIARIO Retorna registros criados em data especifica Sincronização diária
    INCREMENTAL Retorna registros modificados nos últimos N dias Atualizações incrementais
    DINAMICO Escolhe entre FULL/INCREMENTAL baseado em percentual Adaptativo

    Parâmetros:

    Parâmetro Tipo Descrição
    ambiente str Ambiente MongoDB
    qde_dias_retroativos int Dias para buscar modificações

    tp_atualizacao Literal[...] Tipo de estratégia
    nm_collection_implementacao str Collection de implementação
    nm_collection_pacelas str Collection de parcelas
    data_criacao str, optional Data 'YYYY-MM-DD' (obrigatório para DIARIO)

    Retorna: Tuple[str, DataFrame, DataFrame]

    def consultar_colunas_tabela(ambiente:Literal['PRODUCAO','HOMOLOGACAO','ANALYTICS'],nm_tabela:str):
    colunas_tabela = []
    query = r"""
        SELECT column_name
        FROM information_schema.columns
        WHERE table_name = '%s'
        ORDER BY ordinal_position
    """% (nm_tabela)
    try:
        con = abrir_conexao_postgres(ambiente=ambiente)
        cur = con.cursor()
        cur.execute(query=query)

        colunas_tabela = [r[0] for r in cur.fetchall()]
    except Exception as e:
        print( f'Erro : {e}')   

    finally:
        if con:
            con.close()
            cur.close()

    return colunas_tabela

    Funcionalidades:

    • Query INFORMATION_SCHEMA.COLUMNS do PostgreSQL
    • Retorna colunas em ordem (ordinal_position)
    • Utilizada para validação de schema

    nm_tabela str Nome da tabela

        for i, dado in enumerate(lista_uuid, start=1):
            lista_0.append(dado)
            if i % qde_bilhetes == 0:
                lista_fim.append(lista_0)
                lista_0 = []
        if lista_0:
            lista_fim.append(lista_0)

        lista_query = []
        for lista in lista_fim:
            lista_chaves = [f"'{l}'" for l in lista]
            texto_ids = ','.join(lista_chaves)
            lista_query.append(query_delete % texto_ids)

    con = abrir_conexao_postgres(ambiente=ambiente)
    try:
        with con.cursor() as cur:
            if tp_delete.upper() == 'TRUNCATE':
                print(query_truncate)
                cur.execute(query_truncate)
            elif tp_delete.upper() == 'DELETE':
                for query in lista_query:
                    cur.execute(query)
                #print(query)

        con.commit()
    except Exception as e:
        print(f'Erro ao executar DELETE/TRUNCATE na tabela {nm_tabela}: {e}')
    finally:
        con.close()

            cur.close()
        if conn:
            conn.close()
        print(caminho_csv)

    def remover_indices_postgres(ambiente: Literal['PRODUCAO','HOMOLOGACAO','ANALYTICS'], nm_tabela: str):
    """
    Remove todos os índices de uma tabela, exceto índices de PK/FK/unique constraints.
    """
    schema = 'public'  # ajuste se necessário
    conn = abrir_conexao_postgres(ambiente=ambiente)

    try:
        with conn.cursor() as cur:
            # Buscar índices que não são constraints
            cur.execute("""
                SELECT indexname
                FROM pg_indexes
                WHERE schemaname = %s
                AND tablename = %s
                AND indexname NOT IN (
                    SELECT conname
                    FROM pg_constraint
                    WHERE conrelid = %s::regclass
                )
            """, (schema, nm_tabela, nm_tabela))

            indexes = cur.fetchall()

            for idx_name, in indexes:
                print(f"Dropping index {idx_name}...")
                cur.execute(sql.SQL("DROP INDEX IF EXISTS {}").format(sql.Identifier(idx_name)))

        conn.commit()
        print("Todos os índices removidos (exceto constraints).")

    finally:
        conn.close()

    def criar_indices(ambiente: Literal['PRODUCAO','HOMOLOGACAO','ANALYTICS'],indexes: dict, nm_tabela: str):
    """
    Cria índices não-únicos na tabela se não existirem.

    :param indexes: dicionário {nome_do_indice: coluna OU lista de colunas}
    """
    conn = abrir_conexao_postgres(ambiente=ambiente)
    print(f'{fc.data_hora()} Criando índices...')
    try:
        with conn.cursor() as cur:
            for index_name, columns in indexes.items():
                # Garante que columns seja lista
                if isinstance(columns, str):
                    columns = [columns]

                cols_sql = sql.SQL(", ").join(map(sql.Identifier, columns))
                # REMOVE dynamismo do UNIQUE, sempre non-unique
                unique_clause = sql.SQL("")  # sempre vazio

                query = sql.SQL("CREATE {}INDEX IF NOT EXISTS {} ON {} ({})").format(
                    unique_clause,
                    sql.Identifier(index_name),
                    sql.Identifier(nm_tabela),
                    cols_sql
                )

                cur.execute(query)
                print(f"Índice {index_name} criado na tabela {nm_tabela} (não-único).")

        conn.commit()

    finally:
        print(f'{fc.data_hora()} Finalizando criação de índices...')
        conn.close()

    def converter_coluna_para_numerico(df, coluna):
    """
    Converte uma coluna de texto com vírgulas (e opcionalmente separadores de milhar)
    para valores numéricos (float). Valores inválidos são convertidos para NaN.

    Parâmetros:
        df (pd.DataFrame): O DataFrame contendo a coluna.
        coluna (str): O nome da coluna a ser convertida.

    Retorna:
        pd.Series: A coluna convertida para numérico.
    """
    return (
        df[coluna]
        .astype(str)
        .str.replace(r'[^\d,.-]', '', regex=True)  # remove R$, espaços, etc
        #.str.replace('.', '', regex=False)         # remove separador de milhar
        .str.replace(',', '.', regex=False)        # converte decimal vírgula para ponto
        .pipe(pd.to_numeric, errors='coerce')      # converte para número, inválidos viram NaN
    )

    Funcionalidades:

    • Remove simbolos monetários
    • Converte virgula decimal para ponto
    • Preserva separação de milhares
    • Valores invalidos resultam em NaN

    def formatar_datas_coluna_rapido(df, coluna, nova_coluna):
    formatos = [
        "%Y.%m.%d",                 # 2025.02.01
        "%Y-%m-%d",                 # 2025-02-01
        "%d/%m/%Y",                 # 01/02/2025
        "%m/%d/%y",                 # 02/01/2025
        "%d-%m-%Y",                 # 01-02-2025
        "%Y/%m/%d",                 # 2025/02/01
        "%d.%m.%Y",                 # 01.02.2025
        "%Y.%m.%d %H:%M:%S",        # 2024.06.25 05:02:46
        "%Y-%m-%d %H:%M:%S",        # 2024-06-25 05:02:46

        # Formatos ISO 8601
        "%Y-%m-%dT%H:%M:%S.%fZ",    # 2024-06-25T05:02:46.123Z
        "%Y-%m-%dT%H:%M:%S%z",      # 2024-06-25T05:02:46+0000
        "%Y-%m-%dT%H:%M:%S.%f%z",   # 2024-06-25T05:02:46.123+0000
        "%Y-%m-%dT%H:%M:%S",        # 2024-06-25T05:02:46 (sem timezone)
    ]

    resultado = pd.Series(pd.NaT, index=df.index)

    # --- 1) Tratamento para números tipo "serial Excel"
    mask = resultado.isna()
    valores = df.loc[mask, coluna]

    # Identifica valores que podem ser datas Excel:
    # int, float ou string numérica
    possivel_excel = valores.apply(
        lambda x: (isinstance(x, (int, float)) and not pd.isna(x)) or
                (isinstance(x, str) and x.isdigit())
    )

    # Converte strings numéricas para int/float
    serie_numerica = pd.to_numeric(valores[possivel_excel], errors="coerce")

    # Limite seguro para evitar OutOfBoundsDatetime
    limite_superior = 500000  # dias (~ano 3273)
    excel_mask = serie_numerica.between(0, limite_superior)

    # Converter números Excel -> datetime somente se estiverem na faixa segura
    if excel_mask.any():
        resultado.loc[serie_numerica.index[excel_mask]] = pd.to_datetime(
            serie_numerica[excel_mask], 
            unit="D",
            origin="1899-12-30",
            errors="coerce"
        )

    # --- 2) Testar todos os formatos fornecidos
    for fmt in formatos:
        mask = resultado.isna()
        if not mask.any():
            break
        resultado[mask] = pd.to_datetime(df.loc[mask, coluna], format=fmt, errors='coerce')

    # --- 3) Fallback final do pandas (excelente para valores ISO)
    mask = resultado.isna()
    if mask.any():
        resultado[mask] = pd.to_datetime(df.loc[mask, coluna], errors='coerce')
    try:
        df[nova_coluna] = resultado.dt.normalize()
    except:
        df[nova_coluna] = resultado

    return df

    Observacoes Importantes¶

    Boas Práticas

    1. Os pipelines executam etapas especificas conforme necessario (extração, transformação, carga)
    2. As funções PostgreSQL para remoção de indices e importação são utilizadas em pipelines de carga
    3. A validação de schema garante consistencia entre dados processados e estrutura de tabelas
    4. Estratégias de atualização são dinâmicas e adaptaveis
    5. Todas as operações incluem tratamento de erro e logging
